    def mostrar_info_graficas(self):
    """Muestra un cuadro de diálogo con información sobre las gráficas"""
    # Crear el cuadro de diálogo
    msg = QMessageBox()
    msg.setIcon(QMessageBox.Information)
    msg.setWindowTitle("Información sobre las gráficas")
    
    # Configurar el texto con formato HTML
    msg.setText("""
    <style>
    body {
        color: white;
        font-size: 12pt;
    }
    b {
        color: #60D4B8;
    }
    </style>
    <b>Información sobre las gráficas:</b><br><br>
    Los valores mostrados en las gráficas están promediados<br>
    según el filtro de tiempo seleccionado:<br><br>
    • <b>Todo:</b> Promedio general<br>
    • <b>Mes anterior:</b> Promedio del último mes<br>
    • <b>Último trimestre:</b> Promedio de los últimos 3 meses<br>
    • <b>Último semestre:</b> Promedio de los últimos 6 meses<br>
    • <b>Último año:</b> Promedio de los últimos 12 meses<br>
    """)
    
    # Configurar estilo del cuadro de diálogo
    msg.setStyleSheet("""
        QMessageBox {
            background-color: #1f2232;
        }
        QLabel {
            color: white;
        }
        QPushButton {
            background-color: #4A90E2;
            color: white;
            padding: 5px 10px;
            border-radius: 5px;
        }
        QPushButton:hover {
            background-color: #357ABD;
        }
    """)
    
    # Añadir botón de OK
    msg.setStandardButtons(QMessageBox.Ok)
    
    # Mostrar el diálogo
    msg.exec_()
